/** * doGet handles HTML rendering for the parent portal UI * and handles JSON API requests if parameters are provided. */ function doGet(e) { // Check if search query by NIS is requested via API if (e && e.parameter && e.parameter.nis) { try { var studentData = searchStudentByNisBackend(e.parameter.nis); return ContentService.createTextOutput(JSON.stringify(studentData)) .setMimeType(ContentService.MimeType.JSON); } catch(err) { return ContentService.createTextOutput(JSON.stringify({ status: "error", message: err.toString() })) .setMimeType(ContentService.MimeType.JSON); } } // Otherwise, render the Parent Portal UI (Index.html) return HtmlService.createTemplateFromFile('Index') .evaluate() .setTitle('Portal Wali Murid - EUNOIA') .addMetaTag('viewport', 'width=device-width, initial-scale=1') .setXFrameOptionsMode(HtmlService.XFrameOptionsMode.ALLOWALL); } /** * doPost handles the secure sync data payload from EUNOIA React App. * Resolves classic CORS redirection issues by returning plain-text / JSON string. */ function doPost(e) { try { if (!e || !e.postData || !e.postData.contents) { throw new Error("No payload contents found in the POST request."); } var payload = JSON.parse(e.postData.contents); var syncResult = writeDataToSheets(payload); return ContentService.createTextOutput(JSON.stringify({ status: "success", message: "Sync completed successfully!", syncedClasses: syncResult.syncedClasses, totalStudents: syncResult.totalStudents })).setMimeType(ContentService.MimeType.JSON); } catch (error) { return ContentService.createTextOutput(JSON.stringify({ status: "error", message: "Sync failed: " + error.toString() })).setMimeType(ContentService.MimeType.JSON); } } /** * Dynamic sheet recovery or creation helper. */ function getOrCreateSheet(ss, sheetName) { var sheet = ss.getSheetByName(sheetName); if (!sheet) { sheet = ss.insertSheet(sheetName); } return sheet; } /** * Organizes and writes sync payload structured into sheets. */ function writeDataToSheets(data) { var ss = SpreadsheetApp.getActiveSpreadsheet(); // 1. Write metadata / profile sheet var metaSheet = getOrCreateSheet(ss, "PROFIL_SEKOLAH"); metaSheet.clear(); metaSheet.appendRow(["PARAMETER", "VALUE"]); metaSheet.appendRow(["Nama Sekolah", data.schoolName || "-"]); metaSheet.appendRow(["Nama Guru", data.teacherName || "-"]); metaSheet.appendRow(["Mata Pelajaran", data.subject || "-"]); metaSheet.appendRow(["Semester Aktif", data.semester || "-"]); metaSheet.appendRow(["Terakhir Sinkron", new Date().toLocaleString("id-ID")]); metaSheet.getRange("A1:B1").setFontWeight("bold").setBackground("#f1f5f9"); metaSheet.autoResizeColumns(1, 2); // Keep records of classes and count var classCount = 0; var studentTotal = 0; // Prepare database arrays var siswaRows = [["NIS", "NAMA SISWA", "ROMBEL KELAS"]]; var nilaiRows = [["NIS", "NAMA SISWA", "ROMBEL KELAS", "SEMESTER", "MATA PELAJARAN", "RATA FORMATIF (HARIAN)", "RATA SUMATIF (BAB)", "ASTS", "ASAS", "NILAI AKHIR RAPOR"]]; var kehadiranRows = [["NIS", "NAMA SISWA", "ROMBEL KELAS", "SEMESTER", "HADIR (H)", "SAKIT (S)", "IZIN (I)", "ALPA (A)"]]; var studentsDb = data.students || {}; var gradesDb = data.grades || {}; var attendanceDb = data.attendanceList || {}; var weights = (data.gradeConfig && data.gradeConfig.weights) ? data.gradeConfig.weights : { formatif: 30, sumatif: 30, asts: 20, asas: 20 }; var tps = (data.gradeConfig && data.gradeConfig.tps) ? data.gradeConfig.tps : []; var sumatifs = (data.gradeConfig && data.gradeConfig.sumatifs) ? data.gradeConfig.sumatifs : []; for (var className in studentsDb) { classCount++; var studentList = studentsDb[className] || []; for (var i = 0; i < studentList.length; i++) { var s = studentList[i]; studentTotal++; // Push basic info siswaRows.push([s.nis, s.name, className]); // Calculate grades dynamically from active payload var gradeKey = data.semester + "_" + data.subject + "_" + s.id; var g = gradesDb[gradeKey] || {}; var totalF = 0; tps.forEach(function(tp) { totalF += (g[tp.id] || 0); }); var avgF = tps.length > 0 ? totalF / tps.length : 0; var totalS = 0; sumatifs.forEach(function(sm) { totalS += (g[sm.id] || 0); }); var avgS = sumatifs.length > 0 ? totalS / sumatifs.length : 0; var astsVal = g.asts || 0; var asasVal = g.asas || 0; var finalGrade = (avgF * (weights.formatif / 100)) + (avgS * (weights.sumatif / 100)) + (astsVal * (weights.asts / 100)) + (asasVal * (weights.asas / 100)); nilaiRows.push([ s.nis, s.name, className, data.semester, data.subject, Math.round(avgF), Math.round(avgS), astsVal, asasVal, Math.round(finalGrade) ]); // Process attendance records dynamically var hCount = 0, sCount = 0, iCount = 0, aCount = 0; for (var attKey in attendanceDb) { if (attKey.indexOf(className + "_" + data.semester + "_") === 0) { var status = attendanceDb[attKey][s.id]; if (status === 'H') hCount++; if (status === 'S') sCount++; if (status === 'I') iCount++; if (status === 'A') aCount++; } } kehadiranRows.push([ s.nis, s.name, className, data.semester, hCount, sCount, iCount, aCount ]); } } // 2. Write basic students list sheet var siswaSheet = getOrCreateSheet(ss, "DATA_SISWA"); siswaSheet.clear(); siswaSheet.getRange(1, 1, siswaRows.length, siswaRows[0].length).setValues(siswaRows); siswaSheet.getRange("A1:C1").setFontWeight("bold").setBackground("#e2e8f0"); siswaSheet.autoResizeColumns(1, 3); // 3. Write processed grades ledger var nilaiSheet = getOrCreateSheet(ss, "REKAP_NILAI"); nilaiSheet.clear(); nilaiSheet.getRange(1, 1, nilaiRows.length, nilaiRows[0].length).setValues(nilaiRows); nilaiSheet.getRange("A1:J1").setFontWeight("bold").setBackground("#cbd5e1"); nilaiSheet.autoResizeColumns(1, 10); // 4. Write attendance ledger var kehadiranSheet = getOrCreateSheet(ss, "REKAP_KEHADIRAN"); kehadiranSheet.clear(); kehadiranSheet.getRange(1, 1, kehadiranRows.length, kehadiranRows[0].length).setValues(kehadiranRows); kehadiranSheet.getRange("A1:H1").setFontWeight("bold").setBackground("#dfe7f6"); kehadiranSheet.autoResizeColumns(1, 8); return { syncedClasses: classCount, totalStudents: studentTotal }; } /** * Backend fetch function to let parent query records instantly using NIS. * Triggered securely by google.script.run from Index.html */ function searchStudentByNisBackend(nis) { if (!nis) return null; var ss = SpreadsheetApp.getActiveSpreadsheet(); // Fetch from Rekap Nilai & Rekap Kehadiran var nilaiSheet = ss.getSheetByName("REKAP_NILAI"); var hadirSheet = ss.getSheetByName("REKAP_KEHADIRAN"); var profilSheet = ss.getSheetByName("PROFIL_SEKOLAH"); if (!nilaiSheet || !hadirSheet) { return { status: "empty", message: "Database sekolah belum disinkronkan oleh guru pengajar." }; } var schoolName = "Sekolah"; if (profilSheet) { var metaValues = profilSheet.getDataRange().getValues(); for (var m = 0; m < metaValues.length; m++) { if (metaValues[m][0] === "Nama Sekolah") schoolName = metaValues[m][1]; } } var nilaiData = nilaiSheet.getDataRange().getValues(); var studentFound = null; // Search in Grades sheet for (var r = 1; r < nilaiData.length; r++) { if (String(nilaiData[r][0]).trim() === String(nis).trim()) { studentFound = { nis: nilaiData[r][0], name: nilaiData[r][1], className: nilaiData[r][2], semester: nilaiData[r][3], subject: nilaiData[r][4], formatif: nilaiData[r][5], sumatif: nilaiData[r][6], asts: nilaiData[r][7], asas: nilaiData[r][8], finalGrade: nilaiData[r][9], schoolName: schoolName }; break; } } if (!studentFound) { return { status: "not_found", message: "Siswa dengan NIS " + nis + " tidak ditemukan." }; } // Fetch Attendance matching NIS var hadirData = hadirSheet.getDataRange().getValues(); for (var h = 1; h < hadirData.length; h++) { if (String(hadirData[h][0]).trim() === String(nis).trim()) { studentFound.attendance = { hadir: hadirData[h][4], sakit: hadirData[h][5], izin: hadirData[h][6], alpa: hadirData[h][7] }; break; } } return { status: "found", data: studentFound }; }
